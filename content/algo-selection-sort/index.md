def selection_sort(data, n):
	# data = [3, 4, 6, 8, 1, 2]
	for i in range(n):
		min = i
		for j in range(i, n):
			if data[j] < data[min]:
				min = j

		data[i], data[min] = data[min], data[i]

	return data
