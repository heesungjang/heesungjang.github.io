```python
def reverse_words(s: str) -> str:
    words = []
    temp_string = ""

    for char in s.strip():
        if len(temp_string) == 0 and char == " ":
            continue

        if char != " ":
            temp_string += char
            continue
        else:
            words.append(temp_string)
            temp_string = ""
            continue
    words.append(temp_string)
    words.reverse()

    return (" ".join(words))
```
